<tw-amount-currency-select tw-validation
    ng-model="{{ vm.model.components.amountCurrencySelect }}"
    ng-required="{{ vm.amountCurrencySelect.required }}"
    ng-disabled="{{ vm.amountCurrencySelect.disabled }}"
    ng-min="{{ vm.amountCurrencySelect.min }}"
    ng-max="{{ vm.amountCurrencySelect.max }}"
    ng-change="vm.log('amount input value changed:' + vm.model.components.amountCurrencySelect)"
    currency="'{{ vm.model.components.currencySelected }}'"
    currencies="{{ vm.select.options.currencySelect }}"
    on-currency-change="vm.log('currency input value changed:' + vm.model.components.currencySelected)"
    placeholder="{{ vm.amountCurrencySelect.placeholder }}"
    custom-action-label="{{ vm.amountCurrencySelect.customActionLabel }}"
    on-custom-action="vm.log('Custom Action Called')"
    currency-filter-placeholder="{{ vm.amountCurrencySelect.currencyFilterPlaceholder }}"
    size="{{ vm.amountCurrencySelect.size }}"
    lock="{{ vm.amountCurrencySelect.lock }}"
    ></tw-amount-currency-select>
